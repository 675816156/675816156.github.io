INSERTION-SORT(A)

# 伪代码实现:

```c
for j = 1 to A.length
	key = A[j]
	// Insert A[j] into the sorted sequence A[1 .. j - 1].
	i = j - 1
	while i >= 0 and A[i] > key
		A[i + 1] = A[i]
		i = i - 1
	A[i + 1] = key
```

# C代码实现:

```c
#include <stdio.h>
#define NENTS(a) (sizeof(a) / sizeof(a[0]))
void insert_sort(int *a, int n)
{
	int i, j, key;
	for (j = 1; j < n; ++j) {
		key = a[j];
		i = j - 1;
		while (i >= 0 && a[i] > key) {
			a[i + 1] = a[i];
			--i;
		}
		a[i + 1] = key;
	}
}

void show_array(int *a, int n)
{
	int i = 0;
	for (i = 0; i < n; ++i) {
		printf("a[%d] = %d\n", i, a[i]);
	}
}

int main(int argc,  char *argv[])
{
	int a[] = {5, 2, 4, 6, 1, 3};
	show_array(a, NENTS(a));
	insert_sort(a, NENTS(a));
	show_array(a, NENTS(a));
	getchar();
	return 0;
}
```
